98  TChern_v.reserve(binomial(n, s) * binomial(n - s, i));
99  std::vector<char> mono(2 * n);
100  auto c_x = combinations_generator(n, s);
101  auto c_y = combinations_generator(n - s, i);
102  for (const auto& comb_x : c_x) {
103  for (const auto& j : comb_x)
104  mono[j] = 1;
105  std::vector<char> letters_not_in_comb_x;
106  letters_not_in_comb_x.reserve(n - s);
107  for (int j = 0; j < n; j++) {
108  if (std::find(comb_x.begin(), comb_x.end(), j) == comb_x.end())
109  letters_not_in_comb_x.push_back(j);
110  }
111  for (const auto& comb_y : c_y) {
112  for (const auto& j : comb_y)
113  mono[n + letters_not_in_comb_x[j]] = 1;
114  TChern_v.push_back(monomial<scalar_t, halfidempotent>(1, mono));
115  for (const auto& j : comb_y)
116  mono[n + letters_not_in_comb_x[j]] = 0;
117  }
118  for (const auto& j : comb_x)
119  mono[j] = 0;
120  }

152  void set_relations() {
153  std::vector<char> comb(number_of_generators);
154  for (int s = 1; s < n; s++) {
155  for (int i = 1; i <= n - s; i++) {

161  }
162  for (int s = 1; s < n; s++)
163  for (int t = s; t < n; t++)
164  for (int i = std::max(t - s, 1); i <= n - s; i++)
165  for (int j = 1; j <= n - t; j++) {
166  if (s == t && j < i)
167  continue;
168  std::fill(comb.begin(), comb.end(), 0);
169  comb[TwistedChern_indexer.at(std::make_pair(s, i))]++;
170  comb[TwistedChern_indexer.at(std::make_pair(t, j))]++; // TwistedChern_{s,i} TwistedChern_{t,j} is is a relation if s<=t<=s+i
171  relations.push_back(comb);

206  void find_exponent(const monomial<scalar_t, halfidempotent>& term, std::vector<char>& exponent) {
207  if (term.exponent[this->n] > 0) {//clear the consecutive Idem's at the start
208  int consecutive_u_at_start = 0;
209  std::vector<char> us_at_start(2 * this->n);
210  for (int i = this->n; i < term.exponent.size(); i++) {
211  if (term.exponent[i] > 0) {
212  us_at_start[i] = 1;
213  consecutive_u_at_start++;
214  }
215  else
216  break;
217  }
218  if (consecutive_u_at_start > 0)
219  exponent[consecutive_u_at_start-1] = 1;
220  find_exponent(term / monomial<scalar_t, halfidempotent>(1, us_at_start), exponent);
221  return;
222  }
223  bool found_last_u = 0;
224  int last_consecutive_u_from_end = 2 * this->n;
225  int number_consecutive_u = 0;
226  for (int i = 2 * this->n - 1; i >= this->n; i--) {
227  if (term.exponent[i] == 0 && found_last_u)
228  break;
229  else if (term.exponent[i] > 0) {
230  last_consecutive_u_from_end = i;
231  number_consecutive_u++;
232  found_last_u = 1;
233  }
234  }
235  if (last_consecutive_u_from_end == 2 * this->n) {//no Idem's
236  for (int i = 1; i <= this->n; i++) {
237  if (i == this->n)
238  exponent[this->n+i-1] += term.exponent[i - 1];
239  else
240  exponent[this->n + i - 1] += term.exponent[i - 1] - term.exponent[i];
241  }
242  return;
243  }
244  if (last_consecutive_u_from_end > this->n) {//not first Idem
245  //add c_{s,i} to the exponent
246  exponent[this->TwistedChern_indexer[std::make_pair(last_consecutive_u_from_end - this->n, number_consecutive_u)]]++;
247  find_exponent(term / this->TwistedChern[last_consecutive_u_from_end - this->n - 1][number_consecutive_u - 1].highest_term(), exponent);
248  }

252  std::vector<char> pwr_here(this->number_of_generators);
