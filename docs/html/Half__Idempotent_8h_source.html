211  void find_exponent_recursive(const xy_t& term, chern_t& exponent) const {
212  if (term[n] > 0) {//clear the consecutive y_i at the start
213  int consecutive_u_at_start = 0;
214  xy_t us_at_start(2 * n);
215  for (int i = n; i < 2 * n && term[i]>0; i++) {
216  us_at_start[i] = 1;
217  consecutive_u_at_start++;
218  }
219  if (consecutive_u_at_start > 0)
220  exponent[index(0,consecutive_u_at_start)] = 1;
221  find_exponent_recursive(term - us_at_start, exponent);
222  return;
223  }
224  bool found_last_u = 0;
225  int last_consecutive_u_from_end = 2 * n;
226  int number_consecutive_u = 0;
227  for (int i = 2 * n - 1; i >= n; i--) {
228  if (term[i] == 0 && found_last_u)
229  break;
230  else if (term[i] > 0) {
231  last_consecutive_u_from_end = i;
232  number_consecutive_u++;
233  found_last_u = 1;
234  }
235  }
236  if (last_consecutive_u_from_end == 2 * n) {//no y_i means it's an elementary symmetric on the x_i
237  for (int i = 1; i < n; i++)
238  exponent[index(i, 0)] += term[i - 1] - term[i];
239  exponent[index(n, 0)] += term[n - 1];
240  return;
241  }
242  if (last_consecutive_u_from_end > n) {//not first y_i so add c_{s,i} to the exponent
243  auto ind = index(last_consecutive_u_from_end - n, number_consecutive_u);
244  exponent[ind]++;
245  find_exponent_recursive(term - _generators[ind].highest_term().exponent(), exponent);
246  }
247  }
