236  void find_power(const monomial<scalar_t, halfidempotent>& term, std::vector<int>& power) {
237  if (term.powers[this->n] > 0) {//clear the consecutive Idem's at the start
238  int consecutive_u_at_start = 0;
239  std::vector<int> us_at_start(2 * this->n);
240  for (int i = this->n; i < term.powers.size(); i++) {
241  if (term.powers[i] > 0) {
242  us_at_start[i] = 1;
243  consecutive_u_at_start++;
244  }
245  else
246  break;
247  }
248  power[0] = consecutive_u_at_start;
249  find_power(term / monomial<scalar_t, halfidempotent>(1, us_at_start), power);
250  return;
251  }
252  bool found_last_u = 0;
253  int last_consecutive_u_from_end = 2 * this->n;
254  int number_consecutive_u = 0;
255  for (int i = 2 * this->n - 1; i >= this->n; i--) {
256  if (term.powers[i] == 0 && found_last_u)
257  break;
258  else if (term.powers[i] > 0) {
259  last_consecutive_u_from_end = i;
260  number_consecutive_u++;
261  found_last_u = 1;
262  }
263  }
264  if (last_consecutive_u_from_end == 2 * this->n) {//no Idem's
265  for (int i = 1; i <= this->n; i++) {
266  if (i == this->n)
267  power[i] += term.powers[i - 1];
268  else
269  power[i] += term.powers[i - 1] - term.powers[i];
270  }
271  return;
272  }
273  if (last_consecutive_u_from_end > this->n) {//not first Idem
274  //add c_{s,i} to the power
275  power[this->TwistedChern_indexer[std::make_pair(last_consecutive_u_from_end - this->n, number_consecutive_u)]]++;
276  find_power(term / this->TwistedChern[last_consecutive_u_from_end - this->n - 1][number_consecutive_u - 1], power);
277  }
278  }
