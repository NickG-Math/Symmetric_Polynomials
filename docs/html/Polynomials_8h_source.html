168  void condense() {
169  int i = 0;
170  while (i < monos.size()) {
171  if (monos[i].coeff == 0) //erase 0 monomial and dont change i
172  monos.erase(monos.begin() + i);
173  else if (i != monos.size() - 1 && monos[i].degree == monos[i + 1].degree && monos[i].powers == monos[i + 1].powers) {
174  if (monos[i].coeff + monos[i + 1].coeff == 0) { //they cancel out so erase both and dont change i
175  monos.erase(monos.begin() + i, monos.begin() + i + 2);
176  }
177  else {
178  monos[i].coeff += monos[i + 1].coeff; //incorporate to the first and erase the second and dont change i
179  monos.erase(monos.begin() + i + 1);
180  }
181  }
182  else { //nothing erased, change i
183  i++;
184  }
185  }
186  }

275 polynomial<scalar_t, rel_t> multiply(const polynomial<scalar_t, rel_t>& a, const polynomial<scalar_t, rel_t>& b, bool dontcondense = 0) {
276  if (a.monos.empty()) //0 polynomial
277  return a;
278  if (b.monos.empty()) //0 polynomial
279  return b;
280  std::vector<monomial<scalar_t, rel_t>> prodmonos;
281  prodmonos.reserve(a.monos.size() * b.monos.size());
282  for (const auto& i : a.monos)
283  for (const auto& j : b.monos)
284  if (i.coeff != 0 && j.coeff != 0) {
285  if constexpr (rel_t::product_of_monomials_is_monomial)
286  prodmonos.push_back(i * j);
287  else {
288  polynomial<scalar_t, rel_t> ij = i * j;
289  for (const auto& k : ij.monos)
290  prodmonos.push_back(k);
291  }
292  }
293  return polynomial<scalar_t, rel_t>(prodmonos, dontcondense);
294 }
