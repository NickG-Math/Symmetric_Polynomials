291  void find_power(const monomial<scalar_t, halfidempotent>& term, std::vector<int>& power) {
292  if (term.get_powers()[this->n] > 0) {//clear the consecutive Idem's at the start
293  int consecutive_u_at_start = 0;
294  std::vector<int> us_at_start(2 * this->n);
295  for (int i = this->n; i < term.get_powers().size(); i++) {
296  if (term.get_powers()[i] > 0) {
297  us_at_start[i] = 1;
298  consecutive_u_at_start++;
299  }
300  else
301  break;
302  }
303  power[0] = consecutive_u_at_start;
304  find_power(term / monomial<scalar_t, halfidempotent>(1, us_at_start), power);
305  return;
306  }
307  bool found_last_u = 0;
308  int last_consecutive_u_from_end = 2 * this->n;
309  int number_consecutive_u = 0;
310  for (int i = 2 * this->n - 1; i >= this->n; i--) {
311  if (term.get_powers()[i] == 0 && found_last_u)
312  break;
313  else if (term.get_powers()[i] > 0) {
314  last_consecutive_u_from_end = i;
315  number_consecutive_u++;
316  found_last_u = 1;
317  }
318  }
319  if (last_consecutive_u_from_end == 2 * this->n) {//no Idem's
320  for (int i = 1; i <= this->n; i++) {
321  if (i == this->n)
322  power[i] += term.get_powers()[i - 1];
323  else
324  power[i] += term.get_powers()[i - 1] - term.get_powers()[i];
325  }
326  return;
327  }
328  if (last_consecutive_u_from_end > this->n) {//not first Idem
329  //add c_{s,i} to the power
330  power[this->TwistedChern_indexer[std::make_pair(last_consecutive_u_from_end - this->n, number_consecutive_u)]]++;
331  find_power(term / this->TwistedChern[last_consecutive_u_from_end - this->n - 1][number_consecutive_u - 1], power);
332  }
333  }

335  void decompose_recursive(const polynomial<scalar_t, halfidempotent>& a) {
336  std::vector<int> pwr_here(this->dimensions.size());
337  auto max = max_in_orbit(a.highest_term(), this->permutations);
338  find_power(max, pwr_here);
339  powers.push_back(pwr_here);
340  auto product = this->compute_product(pwr_here);
341  auto coeff = max.get_coeff() / product.highest_term().get_coeff();
342  coeffs.push_back(coeff);
343  auto coeff_product = coeff * product;
344  if (a != coeff_product)
345  decompose_recursive(a - coeff_product);
346  }
