10  long binomial(int top, int bot) {
11  //safe implementation
12  if (bot > top - bot)
13  bot = top - bot;
14  int binom = top;
15  for (int i = 1; i <= bot-1; i++) { // $n/1 * (n-1)/2 * \cdots (n-k+1)/k$
16  binom *= (top - i);
17  binom /= i+1;
18  }
19  return binom;
20  }
